#include<iostream>
#include <vector>
#include <sstream>
using namespace std;

int getValue(vector<int> gifts, int n){
    if(n==0)
        return 0;
    int result = gifts[0], count = 1;
    for(int i = 1; i<n; i++){
        if (gifts[i] == result)
            count++;
        else if(count > 0){
            count--;
        }
        else{
            result = gifts[i];
            count = 1;
        }
    } 
    if(count == 0)
        return 0;
    count == 0;
    for(int i = 0; i<n; i++){
        if(gifts[i] == result)
            count++;
    }
    if(count > n/2)
        return result;
    else 
        return 0;
}

int main(){
    vector<int> gifts;
    int itemp;
    int n;
    cin >> n;
    for (int i = 0; i<n; i++){
        cin >> itemp;
        gifts.push_back(itemp);
    }
    cout << getValue(gifts, n) << endl;
}

/*************************************************************************

//对称性
#include<iostream>
#include <string>
#include <vector>
#include <math.h>
using namespace std;

vector<string> getGray(int n){
//    int flag = 0;
//    int step = 0;
    if (n == 1){
        vector<string> gray;
        gray.push_back("0");
        gray.push_back("1");
        return gray;
    }
    else{
        vector<string> last = getGray(n-1);
        int lastsize = last.size();
        int currentsize = 2*lastsize;
        vector<string> current(currentsize);
        for (int i = 0; i<lastsize; i++){
            current[i] = '0'+last[i];
            current[currentsize-1-i] = '1' + last[i];
        }
        return current;
    }
}

int main()
{
    int n;
    while(cin >> n){
        vector<string> gray(getGray(n));
        for(int i = 0; i<pow(2,n); i++){
            cout << gray[i] << endl;
        }
    }
}
