* 数组的翻转

这里用string的翻转为例：

#include<iostream>
#include <algorithm>
#include <stack>
#include <string>
using namespace std;

//c style
void c_rever(string &c, int n)
{
    char temp;
    int j,m = (n-1)/2;
    for(int i=0; i<=m; i++)
    {
        j=n-1-i;
        temp = c[i];
        c[i]=c[j];
        c[j] = temp;
    }
}

int main()
{
    string my= "who are you!";
    int length = my.size();
    c_rever(my, length);
    cout << my << endl;
    reverse(my.begin(), my.end());
    cout << my << endl;

    //stack
    stack<char> s;
    for(int i = 0; i<length; i++)
    {
        s.push(my[i]);
    }
    while(!s.empty()){
        cout << s.top();
        s.pop();
    }
    cout << endl;
    return 0;
}

* 单链表的翻转

Llist list\_reverse(Llist list)
{
    Lnode *tmp = null;
    Lnode *p = null;

    if(list == NULL) return NULL;
    tmp = list->next;
    while(tmp->next != NULL){
        p = tmp->next;
        tmp->next = p->next;
        p->next = list->next;
        list->next = p;
    }
    return list;
}
